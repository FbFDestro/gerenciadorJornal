        let sql = `
        select pes.cpf, pes.nome, pes.tel `;

        }

            sql += `,jor.cpf as jornalista `;

            sql += `,prod.cpf as produtor `;

            sql += `,ed.cpf as editor `;
        }

        if (request.query.pesquisador) {
            sql += `
            inner join pesquisador pesq
            on pes.cpf = pesq.cpf `;
        }
        if (request.query.jornalista) {
            sql += `
            inner join jornalista jor
            on pes.cpf = jor.cpf `;
        }
        if (request.query.produtor) {
            sql += `
            inner join PRODUTOR prod
            on pes.cpf = prod.cpf `;
        }

        if (request.query.editor) {
            sql += `
            inner join editor ed
            on pes.cpf = ed.cpf `;
